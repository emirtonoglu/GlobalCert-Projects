lastaction={}   #dictionary for last action

def timer(a,b):
    if a['auction'] in b:
        time = a['time'] - b[a['auction']]
        b[a['auction']] = a['time']
        return time
    else :
        b[a['auction']] = a['time']
        return 0

bids['timeresponse'] = bids.apply(lambda x: timer(x,lastaction),axis=1)
